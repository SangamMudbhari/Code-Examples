Input/Output Functions

  • Formatted I/O Functions (printf(), scanf())
  • Unformatted I/O Functions (putchar(), getchar(), puts(), gets())
Code Example
#include <stdio.h>

int main() {
    char name[50];
    int age;
    
    printf("Enter name: ");
    gets(name);
    
    printf("Enter age: ");
    scanf("%d", &age);
    
    printf("Hello, %s!\n", name);
    printf("You are %d years old.\n", age);
    return 0;
}
Output
Enter name: John Doe
Enter age: 25
Hello, John Doe!
You are 25 years old.

Control Statements

  • Switch case
  • Goto label
Code Example
int main() {
    int choice = 2;
    
    switch(choice) {
        case 1:
            printf("One");
            break;
        case 2:
            printf("Two");
            break;
        default:
            goto error;
    }
    return 0;

error:
    printf("Invalid choice");
    return 1;
}
Output
Two

Process completed with exit code 0

Code Example
void printPyramid(int n) {
    for(int i = 0; i < n; i++) {
        for(int j = 0; j <= i; j++) {
            printf("* ");
        }
        printf("\n");
    }
}

* * *

Code Example
                  #include <stdio.h>
                  #include <string.h>
                  
                    
                    int main() {
                        char str1[] = "Hello, World!";
                        printf("Length of string \"%s\": %zu\n", str1, strlen(str1));
                    
                        char str2[] = "HELLO";
                        for (int i = 0; str2[i]; i++) {
                            str2[i] = tolower(str2[i]);
                        }
                        printf("Lowercase string: %s\n", str2);
                    
                        char str3[] = "world";
                        for (int i = 0; str3[i]; i++) {
                            str3[i] = toupper(str3[i]);
                        }
                        printf("Uppercase string: %s\n", str3);
                    
                        char str4[50];
                        strcpy(str4, "Copy this string");
                        printf("Copied string: %s\n", str4);
                    
                        char str5[] = "apple";
                        char str6[] = "banana";
                        int result = strcmp(str5, str6);
                        if (result < 0) {
                            printf("\"%s\" is less than \"%s\"\n", str5, str6);
                        } else if (result == 0) {
                            printf("\"%s\" is equal to \"%s\"\n", str5, str6);
                        } else {
                            printf("\"%s\" is greater than \"%s\"\n", str5, str6);
                        }
                    
                        char str7[50] = "Hello, ";
                        char str8[] = "world!";
                        strcat(str7, str8);
                        printf("Concatenated string: %s\n", str7);
                    
                        char str9[50] = "Hello, ";
                        char str10[] = "world! It's a nice day.";
                        strncat(str9, str10, 5);
                        printf("Concatenated with limit: %s\n", str9);
                    
                        return 0;
                    }
Output
 Length of string "Hello, World!": 13
                  Lowercase string: hello
                  Uppercase string: WORLD
                  Copied string: Copy this string
                  "apple" is less than "banana"
                  Concatenated string: Hello, world!
                  Concatenated with limit: Hello, world
